Config

desc A thin wrapper for Config with support for environments, databases, encryption
date 2016-7-5 6:4:25
version 0.9.1
namespace slate.core.common
source slate.core.common.Conf
example Example_Config

Import

// required 
import slate.common.databases.DbConString
import slate.common.encrypt.Encryptor
import slate.core.common.Conf



// optional 
import slate.core.cmds.Cmd
import slate.common.results.ResultSupportIn

Setup


n/a

Usage



    // CASE 1: Load up config from application.conf
    val conf = new Conf()
    conf.init()
    println( "env.name: " + conf.getString("env.name") )
    println( "db.enabled: " + conf.getBool("db.enabled") )
    println()

    // CASE 2: Get the environment selection ( env, dev, qa ) from conf.
    val env = conf.env()
    println( s"${env.name}, ${env.env}, ${env.key}")
    println()

    // CASE 3: Get the db connection info from conf
    // Use this mode if you plan on only having 1 db connection info in the config.
    // NOTE: Refer to src\main\resources\configSample1.conf
    val conf1 = new Conf()
    conf1.init(file = Some("configSample1.conf"))
    val dbCon = conf1.dbCon()
    printDbCon ( "db con : ", dbCon )

    // CASE 4: Get the db connection info from conf with environment prefix
    // Use this mode if you have db settings in the conf file for each environment.
    val conf2 = new Conf()
    conf2.init(file = Some("configSample2.conf"))
    val dbConEnv = conf2.dbCon(enableEnvPrefix = Some(true))
    printDbCon ( "db con with env prefix: ", dbConEnv )

    // CASE 5: Get the db connection string from a settings file called "db-lcl.txt"
    // in a folder called ".slate" in the user directory. This is for added security to avoid
    // any database connection strings in the application.conf file.
    val conf3 = new Conf()
    conf3.init(file = Some("configSample3.conf"))
    val dbConUserDir = conf3.dbCon(userFolder = Some(".slate"))
    printDbCon ( "db con from user folder: ", dbConUserDir )

    // CASE 6: Get the db connection string from a settings file called "db-lcl.txt"
    // in a folder called ".slate" in the user directory. This is for added security to avoid
    // any database connection strings in the application.conf file.
    // This is same as #5 but with the environment prefix!
    val conf4 = new Conf()
    conf4.init(file = Some("configSample4.conf"))
    val dbConUserDirWithEnv = conf4.dbCon(enableEnvPrefix = Some(true), userFolder = Some(".slate"))
    printDbCon ( "db con with env prefix and user folder: ", dbConUserDirWithEnv )

    // CASE 7: Enable encrypted strings in the config file
    val encryptor = new Encryptor("wejklhviuxywehjk", "3214maslkdf03292")
    val conf5 = new Conf()
    conf5.init( file = Some("configSample5.conf"), encryptor = Some(encryptor) )
    println ( "db user decrypted : " + conf5.getStringEnc("db.user") )
    println ( "db pswd decrypted : " + conf5.getStringEnc("db.pswd") )
    println()
